""" Reads the most recent trend_report_*.csv (produced by main.py) and generates an HTML dashboard you can open in your browser. Run: python3 generate_dashboard.py Then: open trend_dashboard.html (macOS) Shows what we actually have data for: trend score, Amazon rank source, price, rating, Trends momentum. It deliberately does NOT show a "where to buy" or "predicted sales" column with made-up numbers — those need the price-match layer and a real sales-velocity source (e.g. Keepa) that don't exist in this pipeline yet. Building fake figures into a dashboard would make it look more trustworthy than it is. """ import csv import glob import json import os import sys def find_latest_report() -> str: files = glob.glob("trend_report_*.csv") if not files: print("No trend_report_*.csv found in this folder.") print("Run 'python main.py' first to generate one.") sys.exit(1) return max(files, key=os.path.getmtime) def load_rows(path: str) -> list[dict]: with open(path, newline="", encoding="utf-8") as f: return list(csv.DictReader(f)) def build_html(rows: list[dict], source_file: str) -> str: # Keep only rows with a title, top 40 by trend_score for readability clean = [r for r in rows if r.get("title")] clean.sort(key=lambda r: float(r.get("trend_score") or 0), reverse=True) top = clean[:40] chart_labels = json.dumps([r["title"][:40] for r in top[:15]]) chart_scores = json.dumps([float(r.get("trend_score") or 0) for r in top[:15]]) table_rows = "\n".join( f""" {r.get('trend_score','')} {r.get('category','')} {r.get('source','')} {r.get('rank','')} {r.get('title','')} £{r.get('price_gbp','') or '-'} {r.get('momentum_pct','') or '-'}% {r.get('bought_last_month') or 'no badge / not checked'} Not yet — price-match layer needed """ for r in top ) return f""" Trend Layer Dashboard

Trend Layer Dashboard

Source file: {source_file} — top {len(top)} products by trend_score
What this shows: Amazon UK Best Sellers/Movers rank + Google Trends UK momentum + Amazon's own disclosed "bought in past month" badge (real data, but bucketed/rounded — e.g. "600+" — and only checked for a capped number of top candidates to respect Amazon's rate limits). What it doesn't show yet: where to source each product locally — that needs the price-match layer (Tesco/Asda/TK Maxx lookup), which doesn't exist in this pipeline yet.
{table_rows}
Trend ScoreCategorySourceRank TitlePriceMomentum Bought last monthWhere to buy
""" def run(): report_file = find_latest_report() rows = load_rows(report_file) html = build_html(rows, report_file) with open("trend_dashboard.html", "w", encoding="utf-8") as f: f.write(html) print(f"Built trend_dashboard.html from {report_file}") print("Open it with: open trend_dashboard.html") if __name__ == "__main__": run()
